isort: GitHub Actions Workflow

Add this workflow file to .github/workflows/isort.yml:


name: isort check

on: [push, pull_request]

jobs:
  isort:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.x'
      - name: Install isort
        run: pip install isort
      - name: Run isort check
        run: isort . --check --diff
